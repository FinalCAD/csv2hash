def check_params rule
  if OLD_RULES_NAME.include? rule
    warn "[DEPRECATION]: `#{rule}` is deprecated.  Please use `#{NEW_SYNTAX[rule]}` instead."
    NEW_SYNTAX[rule]
  else
    rule.to_s
  end
end
